
SELECT c.login AS courier_login, COUNT(o.id) AS delivery_orders_count
FROM "Couriers" AS c
LEFT JOIN "Orders" AS o ON c.id = o."courierId"
WHERE "inDelivery" = true
GROUP BY c.login;


SELECT
    "Orders",
    CASE
        WHEN finished = true THEN 2
        WHEN cancelled = true THEN -1
        WHEN "inDelivery" = true THEN 1
        ELSE 0
    END AS order_status
FROM
    "Orders";
